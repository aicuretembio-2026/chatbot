## 🔧 **기술적 구현**

### **1. 동적 스크립트 로딩**

```javascript
// 언어별 챗봇 데이터 동적 로딩
async function loadChatbotData(lang) {
    if (lang === 'ko') {
        currentChatbotData = CHATBOT_DATA; // 기본 파일
    } else {
        // translations/{lang}/chatbot-data.js 동적 로드
        const script = document.createElement('script');
        script.src = `translations/${lang}/chatbot-data.js`;
        document.body.appendChild(script);
    }
}
```

### **2. UI 텍스트 자동 업데이트**

```javascript
function updateChatbotUI(lang) {
    const translations = CHATBOT_I18N[lang] || CHATBOT_I18N['ko'];
    
    // data-i18n 속성이 있는 모든 요소 업데이트
    document.querySelectorAll('[data-i18n]').forEach(element => {
        const key = element.getAttribute('data-i18n');
        if (translations[key]) {
            element.innerHTML = translations[key];
        }
    });
    
    // 페이지 타이틀 업데이트
    document.title = `${translations.page_title} - ${translations.page_subtitle}`;
}
```

### **3. 언어 저장 및 복원**

// 초기 로드 시

changeChatbotLanguage(savedLang);

localStorage.setItem('chatbotLanguage', selectedLang);
